កម្មវិធីចុះកម្មសិក្សារបស់និស្សិត

ក្នុងអំឡុងឆមាសចុងក្រោយ នៃឆ្នាំបញ្ចប់ និស្សិតនឹងត្រូវចូលរួមក្នុងបរិស្ថានការងារ ដែលរំពឹងថា គេនឹងចាប់យកនូវសកម្មភាពផ្សេងៗ នៅក្នុងអង្គការ ដែលនិស្សិតទាំងនោះទៅរួមការងារជាមួយ។ កម្មវិធីចុះកម្មសិក្សារបស់និស្សិត គឺជាផ្នែកមួយផងដែរក្នុងកម្មវិធីសិក្សារបស់វិទ្យាស្ថាន។ ការចូលរួម ៦ សប្តាហ៍ ក្នុងបរិស្ថានការងារ មិនត្រឹមតែជួយជម្រុញឲ្យនិស្សិតយល់ពីការអនុវត្តន៍នូវមុខវិជ្ជា ចំជំនាញ ជាមួយនឹងអង្គភាពប៉ុណ្ណោះទេ ប៉ុន្តែក៏ផ្តល់ឱកាសឲ្យពួកគេរីកចម្រើនទៅជាមនុស្សធំទទួលខុសត្រូវការងារ។ កម្មវិធីចុះកម្មសិក្សារបស់និស្សិត ផ្តល់ដល់និស្សិតនូវកាលានុវត្តិភាព ដើម្បីអនុវត្ត និងប្រមូលផ្តុំ នូវចំណេះដឹង និងជំនាញដែលទទួលបានក្នុងអំឡុងពេលនៃការសិក្សារបស់គេ ។ និស្សិត នឹងត្រូវគេប្រគល់ការងារ ដែលតម្រូវឲ្យបង្ហាញនូវភាពឯករាជ្យ គំនិតផ្តួចផ្តើម ភាពច្នៃប្រឌិតការបណ្តុះគំនិត និង ភាពប៉ិនប្រសប់។
